    <script type="text/babel">
        function StockMarket() {
            const initialStocks = [
                { name: "TCS", price: 3500, prevPrice: 3500 },
                { name: "Reliance", price: 2700, prevPrice: 2700 },
                { name: "Infosys", price: 1600, prevPrice: 1600 },
                { name: "HDFC Bank", price: 1500, prevPrice: 1500 },
                { name: "ICICI Bank", price: 930, prevPrice: 930 },
                { name: "Tata Motors", price: 670, prevPrice: 670 },
                { name: "SBI", price: 600, prevPrice: 600 },
                { name: "HUL", price: 2500, prevPrice: 2500 },
                { name: "Bharti Airtel", price: 1150, prevPrice: 1150 },
                { name: "Asian Paints", price: 2800, prevPrice: 2800 },
                { name: "ITC", price: 420, prevPrice: 420 },
                { name: "L&T", price: 3200, prevPrice: 3200 },
                { name: "Axis Bank", price: 1020, prevPrice: 1020 },
                { name: "Maruti Suzuki", price: 10500, prevPrice: 10500 },
                { name: "Wipro", price: 450, prevPrice: 450 },
                { name: "Sun Pharma", price: 1400, prevPrice: 1400 },
                { name: "Bajaj Finance", price: 6800, prevPrice: 6800 },
                { name: "NTPC", price: 280, prevPrice: 280 },
                { name: "Power Grid", price: 240, prevPrice: 240 },
                { name: "Tech Mahindra", price: 1250, prevPrice: 1250 },
                { name: "Hero MotoCorp", price: 3100, prevPrice: 3100 },
                { name: "Titan", price: 3400, prevPrice: 3400 },
                { name: "Nestle India", price: 24500, prevPrice: 24500 },
                { name: "Coal India", price: 320, prevPrice: 320 },
                { name: "UltraTech Cement", price: 8900, prevPrice: 8900 },
                { name: "Dr Reddy's", price: 5600, prevPrice: 5600 },
                { name: "Adani Ports", price: 850, prevPrice: 850 },
                { name: "Grasim", price: 2100, prevPrice: 2100 },
                { name: "JSW Steel", price: 840, prevPrice: 840 },
                { name: "Tata Steel", price: 130, prevPrice: 130 },
                { name: "M&M", price: 1650, prevPrice: 1650 },
                { name: "Bajaj Auto", price: 7200, prevPrice: 7200 },
                { name: "Cipla", price: 1180, prevPrice: 1180 },
                { name: "BPCL", price: 460, prevPrice: 460 },
                { name: "IndusInd Bank", price: 1480, prevPrice: 1480 },
                { name: "Britannia", price: 4600, prevPrice: 4600 },
                { name: "ONGC", price: 210, prevPrice: 210 },
                { name: "Eicher Motors", price: 3850, prevPrice: 3850 },
                { name: "Tata Consumer", price: 960, prevPrice: 960 },
                { name: "Hindalco", price: 520, prevPrice: 520 }
            ];
        
            // Initialize stocks from localStorage or use initial values
            const [stocks, setStocks] = React.useState(() => {
                try {
                    const savedStocks = localStorage.getItem('stocks');
                    return savedStocks ? JSON.parse(savedStocks) : initialStocks;
                } catch (error) {
                    console.error("Error loading stocks from localStorage:", error);
                    return initialStocks;
                }
            });
        
            const [searchTerm, setSearchTerm] = React.useState('');
            const [wallet, setWallet] = React.useState(() => {
                try {
                    const savedWallet = localStorage.getItem('wallet');
                    return savedWallet ? Number(savedWallet) : 100000;
                } catch (error) {
                    console.error("Error loading wallet from localStorage:", error);
                    return 100000;
                }
            });
            const [selectedStock, setSelectedStock] = React.useState(null);
            const [quantity, setQuantity] = React.useState(1);
            const [portfolio, setPortfolio] = React.useState(() => {
                try {
                    const savedPortfolio = localStorage.getItem('portfolio');
                    return savedPortfolio ? JSON.parse(savedPortfolio) : [];
                } catch (error) {
                    console.error("Error loading portfolio from localStorage:", error);
                    return [];
                }
            });
            const [watchlist, setWatchlist] = React.useState(() => {
                try {
                    const savedWatchlist = localStorage.getItem('watchlist');
                    return savedWatchlist ? JSON.parse(savedWatchlist) : [];
                } catch (error) {
                    console.error("Error loading watchlist from localStorage:", error);
                    return [];
                }
            });
        
            // Price updates effect
            React.useEffect(() => {
                const timer = setInterval(() => {
                    setStocks(prevStocks => {
                        try {
                            const updatedStocks = prevStocks.map(stock => {
                                const prevPrice = stock.price;
                                let change = 0;
                                
                                if (stock.price < 1000) change = Math.random() * 5;
                                else if (stock.price < 2000) change = Math.random() * 10;
                                else if (stock.price < 5000) change = Math.random() * 25;
                                else if (stock.price < 10000) change = Math.random() * 50;
                                else change = Math.random() * 100;
                                
                                const direction = Math.random() < 0.5 ? -1 : 1;
                                const newPrice = Math.max(stock.price + direction * change, 1);
                                
                                return {
                                    ...stock,
                                    prevPrice,
                                    price: parseFloat(newPrice.toFixed(2))
                                };
                            });
                            
                            localStorage.setItem('stocks', JSON.stringify(updatedStocks));
                            return updatedStocks;
                        } catch (error) {
                            console.error("Error updating stock prices:", error);
                            return prevStocks;
                        }
                    });
                }, 2000);
        
                return () => clearInterval(timer);
            }, []);  
        
            // Update portfolio when stock prices change
            React.useEffect(() => {
                if (portfolio.length > 0) {
                    try {
                        const updatedPortfolio = portfolio.map(position => {
                            const currentStock = stocks.find(s => s.name === position.name);
                            if (currentStock) {
                                return {
                                    ...position,
                                    currentPrice: currentStock.price
                                };
                            }
                            return position;
                        });
                        
                        setPortfolio(updatedPortfolio);
                    } catch (error) {
                        console.error("Error updating portfolio with new prices:", error);
                    }
                }
            }, [stocks, portfolio.length]);
        
            // localStorage updates effect
            React.useEffect(() => {
                try {
                    localStorage.setItem('wallet', wallet.toString());
                    localStorage.setItem('portfolio', JSON.stringify(portfolio));
                    localStorage.setItem('watchlist', JSON.stringify(watchlist));
                } catch (error) {
                    console.error("Error saving to localStorage:", error);
                }
            }, [wallet, portfolio, watchlist]);
        
            // Buying stocks
            const buyStock = (stock) => {
                try {
                    if (!stock) {
                        alert('Please select a stock to buy.');
                        return;
                    }
                    
                    const totalCost = stock.price * quantity;
                    if (totalCost > wallet) {
                        alert('Insufficient funds!');
                        return;
                    }
        
                    // Update wallet
                    const newWallet = wallet - totalCost;
                    setWallet(newWallet);
        
                    // Update portfolio
                    const existingPosition = portfolio.find(p => p.name === stock.name);
                    let updatedPortfolio;
        
                    if (existingPosition) {
                        const totalShares = existingPosition.quantity + quantity;
                        updatedPortfolio = portfolio.map(p => 
                            p.name === stock.name 
                                ? { 
                                    ...p, 
                                    quantity: totalShares,
                                    buyPrice: parseFloat(((p.buyPrice * p.quantity + stock.price * quantity) / totalShares).toFixed(2)),
                                    currentPrice: stock.price
                                }
                                : p
                        );
                    } else {
                        updatedPortfolio = [...portfolio, { 
                            name: stock.name, 
                            quantity, 
                            buyPrice: parseFloat(stock.price.toFixed(2)),
                            currentPrice: stock.price
                        }];
                    }
        
                    setPortfolio(updatedPortfolio);
                    alert(`Successfully bought ${quantity} shares of ${stock.name}`);
                } catch (error) {
                    console.error("Error buying stock:", error);
                    alert('An error occurred while buying stock.');
                }
            };
        
            // Selling stocks
            const sellStock = (stock) => {
                try {
                    if (!stock) {
                        alert('Please select a stock to sell.');
                        return;
                    }
                    
                    const position = portfolio.find(p => p.name === stock.name);
                    if (!position) {
                        alert(`You don't own any shares of ${stock.name}!`);
                        return;
                    }
                    
                    if (position.quantity < quantity) {
                        alert(`You only have ${position.quantity} shares to sell!`);
                        return;
                    }
        
                    const saleValue = stock.price * quantity;
                    
                    // Update wallet
                    const newWallet = wallet + saleValue;
                    setWallet(newWallet);
        
                    // Update portfolio
                    const updatedPortfolio = portfolio.map(p => 
                        p.name === stock.name 
                            ? { ...p, quantity: p.quantity - quantity }
                            : p
                    ).filter(p => p.quantity > 0);
                    
                    setPortfolio(updatedPortfolio);
                    alert(`Successfully sold ${quantity} shares of ${stock.name}`);
                } catch (error) {
                    console.error("Error selling stock:", error);
                    alert('An error occurred while selling stock.');
                }
            };
        
            // Toggle watchlist
            const toggleWatchlist = (stockName) => {
                try {
                    if (!stockName) return;
                    
                    if (watchlist.includes(stockName)) {
                        setWatchlist(watchlist.filter(name => name !== stockName));
                    } else {
                        setWatchlist([...watchlist, stockName]);
                    }
                } catch (error) {
                    console.error("Error toggling watchlist:", error);
                }
            };
        
            // Reset market
            const resetMarket = () => {
                try {
                    if (window.confirm("Are you sure you want to reset the market? This will reset all prices to their initial values.")) {
                        setStocks(initialStocks);
                        localStorage.setItem('stocks', JSON.stringify(initialStocks));
                        alert("Market has been reset successfully.");
                    }
                } catch (error) {
                    console.error("Error resetting market:", error);
                    alert('An error occurred while resetting the market.');
                }
            };
        
            // Reset everything
            const resetEverything = () => {
                try {
                    if (window.confirm("Are you sure you want to reset everything? This will clear your portfolio, watchlist, and reset your wallet.")) {
                        setStocks(initialStocks);
                        setWallet(100000);
                        setPortfolio([]);
                        setWatchlist([]);
                        localStorage.setItem('stocks', JSON.stringify(initialStocks));
                        localStorage.setItem('wallet', '100000');
                        localStorage.setItem('portfolio', '[]');
                        localStorage.setItem('watchlist', '[]');
                        alert("All data has been reset successfully.");
                    }
                } catch (error) {
                    console.error("Error resetting everything:", error);
                    alert('An error occurred while resetting the data.');
                }
            };
        
            // Calculate total portfolio value
            const calculatePortfolioValue = () => {
                return portfolio.reduce((total, position) => {
                    return total + (position.currentPrice * position.quantity);
                }, 0);
            };
        
            // Calculate total profit/loss
            const calculateTotalProfitLoss = () => {
                return portfolio.reduce((total, position) => {
                    const positionValue = position.currentPrice * position.quantity;
                    const costBasis = position.buyPrice * position.quantity;
                    return total + (positionValue - costBasis);
                }, 0);
            };
        
            // Handle quantity change
            const handleQuantityChange = (e) => {
                const value = parseInt(e.target.value);
                if (isNaN(value) || value < 1) {
                    setQuantity(1);
                } else {
                    setQuantity(value);
                }
            };
        
            // Filter stocks based on search term
            const filteredStocks = stocks.filter(stock => 
                stock.name.toLowerCase().includes(searchTerm.toLowerCase())
            );
        
            // Calculate portfolio metrics
            const portfolioValue = calculatePortfolioValue();
            const totalProfitLoss = calculateTotalProfitLoss();
            const totalProfitLossPercent = portfolio.length > 0 
                ? (totalProfitLoss / (portfolioValue - totalProfitLoss)) * 100 
                : 0;
        
            return (
                <div className="market-container w-full max-w-6xl mx-auto my-8 px-4">
                    <h1 className="text-4xl font-bold text-slate-800 mb-6">Stock Market Simulator</h1>
                    
                    {/* Dashboard */}
                    <div className="grid grid-cols-1 md:grid-cols-2 gap-6 mb-8">
                        <div className="bg-gradient-to-r from-indigo-600 to-indigo-500 text-white p-6 rounded-xl shadow-lg">
                            <h2 className="text-lg opacity-90 mb-1">Cash Balance</h2>
                            <p className="text-3xl font-semibold">₹{wallet.toLocaleString('en-IN', { maximumFractionDigits: 2 })}</p>
                        </div>
                        
                        <div className={`bg-gradient-to-r ${totalProfitLoss >= 0 ? 'from-green-600 to-green-500' : 'from-red-600 to-red-500'} text-white p-6 rounded-xl shadow-lg`}>
                            <h2 className="text-lg opacity-90 mb-1">Portfolio Value</h2>
                            <p className="text-3xl font-semibold">₹{portfolioValue.toLocaleString('en-IN', { maximumFractionDigits: 2 })}</p>
                            {portfolio.length > 0 && (
                                <p className="mt-2">
                                    {totalProfitLoss >= 0 ? '+' : ''}₹{totalProfitLoss.toLocaleString('en-IN', { maximumFractionDigits: 2 })} 
                                    ({totalProfitLossPercent >= 0 ? '+' : ''}{totalProfitLossPercent.toFixed(2)}%)
                                </p>
                            )}
                        </div>
                    </div>
        
                    <div className="flex flex-col md:flex-row justify-between items-start md:items-center mb-8 gap-4">
                        <input 
                            type="text"
                            placeholder="Search stocks..."
                            value={searchTerm}
                            onChange={(e) => setSearchTerm(e.target.value)}
                            className="w-full md:flex-1 px-6 py-4 text-lg border-2 border-slate-200 rounded-xl focus:outline-none focus:border-indigo-500 transition-colors shadow-sm"
                        />
                        <div className="flex gap-4 w-full md:w-auto">
                            <button
                                onClick={resetMarket}
                                className="w-full md:w-auto px-6 py-4 bg-slate-200 text-slate-700 font-medium rounded-xl hover:bg-slate-300 transition-colors"
                            >
                                Reset Market
                            </button>
                            <button
                                onClick={resetEverything}
                                className="w-full md:w-auto px-6 py-4 bg-red-100 text-red-700 font-medium rounded-xl hover:bg-red-200 transition-colors"
                            >
                                Reset Everything
                            </button>
                        </div>
                    </div>
        
                    {/* Portfolio Section */}
                    {portfolio.length > 0 && (
                        <div className="mb-8">
                            <h2 className="text-2xl font-bold text-slate-800 mb-4">Your Portfolio</h2>
                            <div className="bg-white rounded-xl border border-slate-200 p-6 shadow-sm">
                                <div className="overflow-x-auto">
                                    <table className="w-full">
                                        <thead>
                                            <tr className="border-b border-slate-200">
                                                <th className="text-left py-3 px-4 text-slate-500 font-medium">Stock</th>
                                                <th className="text-right py-3 px-4 text-slate-500 font-medium">Quantity</th>
                                                <th className="text-right py-3 px-4 text-slate-500 font-medium">Avg. Buy Price</th>
                                                <th className="text-right py-3 px-4 text-slate-500 font-medium">Current Price</th>
                                                <th className="text-right py-3 px-4 text-slate-500 font-medium">Total Value</th>
                                                <th className="text-right py-3 px-4 text-slate-500 font-medium">Profit/Loss</th>
                                            </tr>
                                        </thead>
                                        <tbody>
                                            {portfolio.map(position => {
                                                const totalValue = position.currentPrice * position.quantity;
                                                const profitLoss = (position.currentPrice - position.buyPrice) * position.quantity;
                                                const profitLossPercent = ((position.currentPrice - position.buyPrice) / position.buyPrice) * 100;
                                                
                                                return (
                                                    <tr key={position.name} className="border-b border-slate-100 hover:bg-slate-50">
                                                        <td className="py-3 px-4 font-medium">{position.name}</td>
                                                        <td className="py-3 px-4 text-right">{position.quantity}</td>
                                                        <td className="py-3 px-4 text-right">₹{position.buyPrice.toFixed(2)}</td>
                                                        <td className="py-3 px-4 text-right">₹{position.currentPrice.toFixed(2)}</td>
                                                        <td className="py-3 px-4 text-right">₹{totalValue.toFixed(2)}</td>
                                                        <td className={`py-3 px-4 text-right font-medium ${profitLoss >= 0 ? 'text-green-500' : 'text-red-500'}`}>
                                                            {profitLoss >= 0 ? '+' : ''}₹{profitLoss.toFixed(2)} ({profitLossPercent >= 0 ? '+' : ''}{profitLossPercent.toFixed(2)}%)
                                                        </td>
                                                    </tr>
                                                );
                                            })}
                                        </tbody>
                                    </table>
                                </div>
                            </div>
                        </div>
                    )}
        
                    {/* Watchlist Section */}
                    {watchlist.length > 0 && (
                        <div className="mb-8">
                            <h2 className="text-2xl font-bold text-slate-800 mb-4">Your Watchlist</h2>
                            <div className="space-y-4">
                                {stocks
                                    .filter(stock => watchlist.includes(stock.name))
                                    .map(stock => (
                                        <div 
                                            key={stock.name}
                                            className={`bg-white rounded-xl border border-slate-200 p-6 shadow-sm transition-all duration-200
                                                ${selectedStock?.name === stock.name ? 'ring-2 ring-indigo-500' : 'hover:shadow-md'}`}
                                            onClick={() => setSelectedStock(stock)}
                                        >
                                            <div className="flex justify-between items-center">
                                                <div className="flex items-center space-x-4">
                                                    <h3 className="text-xl font-semibold text-slate-800">{stock.name}</h3>
                                                    <span className="text-blue-500">★</span>
                                                </div>
                                                
                                                <div className="flex items-center space-x-6">
                                                    <div className="flex flex-col items-end">
                                                        <span className={`text-lg font-medium ${
                                                            stock.price > stock.prevPrice ? 'text-green-500' : 'text-red-500'
                                                        }`}>
                                                            ₹{stock.price.toFixed(2)}
                                                        </span>
                                                        <span className={`text-sm ${
                                                            stock.price > stock.prevPrice ? 'text-green-500' : 'text-red-500'
                                                        }`}>
                                                            {stock.price > stock.prevPrice ? '+' : ''}
                                                            {(((stock.price - stock.prevPrice) / stock.prevPrice) * 100).toFixed(2)}%
                                                        </span>
                                                    </div>
        
                                                    {selectedStock?.name === stock.name && (
                                                        <div className="flex items-center space-x-3">
                                                            <input
                                                                type="number"
                                                                min="1"
                                                                value={quantity}
                                                                onChange={handleQuantityChange}
                                                                className="w-24 px-3 py-2 border-2 border-slate-200 rounded-lg focus:outline-none focus:border-indigo-500"
                                                                onClick={e => e.stopPropagation()}
                                                            />
                                                            <button 
                                                                onClick={e => {
                                                                    e.stopPropagation();
                                                                    buyStock(stock);
                                                                }}
                                                                className="px-6 py-2 bg-indigo-600 text-white font-medium rounded-lg hover:bg-indigo-700 transition-colors"
                                                            >
                                                                Buy
                                                            </button>
                                                            <button 
                                                                onClick={e => {
                                                                    e.stopPropagation();
                                                                    sellStock(stock);
                                                                }}
                                                                className="px-6 py-2 border-2 border-red-500 text-red-500 font-medium rounded-lg hover:bg-red-50 transition-colors"
                                                            >
                                                                Sell
                                                            </button>
                                                            <button
                                                                onClick={e => {
                                                                    e.stopPropagation();
                                                                    toggleWatchlist(stock.name);
                                                                }}
                                                                className="px-6 py-2 border-2 border-slate-200 text-slate-600 font-medium rounded-lg hover:bg-slate-50 transition-colors"
                                                            >
                                                                Unwatch
                                                            </button>
                                                        </div>
                                                    )}
                                                </div>
                                            </div>
        
                                            {selectedStock?.name === stock.name && portfolio.find(p => p.name === stock.name) && (
                                                <div className="mt-4 p-4 bg-slate-50 rounded-lg">
                                                    <div className="text-slate-600">
                                                        Your position: {portfolio.find(p => p.name === stock.name).quantity} shares
                                                        @ avg. ₹{portfolio.find(p => p.name === stock.name).buyPrice.toFixed(2)}
                                                    </div>
                                                </div>
                                            )}
                                        </div>
                                    ))}
                            </div>
                        </div>
                    )}
        
                    <h2 className="text-2xl font-bold text-slate-800 mb-4">All Stocks</h2>
                    <div className="space-y-4">
                        {filteredStocks.length > 0 ? (
                            filteredStocks.map(stock => (
                                <div 
                                    key={stock.name}
                                    className={`bg-white rounded-xl border border-slate-200 p-6 shadow-sm transition-all duration-200
                                        ${selectedStock?.name === stock.name ? 'ring-2 ring-indigo-500' : 'hover:shadow-md'}`}
                                    onClick={() => setSelectedStock(stock)}
                                >
                                    <div className="flex flex-col md:flex-row md:justify-between md:items-center">
                                        <div className="flex items-center space-x-4 mb-4 md:mb-0">
                                            <h3 className="text-xl font-semibold text-slate-800">{stock.name}</h3>
                                            {watchlist.includes(stock.name) && (
                                                <span className="text-blue-500">★</span>
                                            )}
                                        </div>
                                        
                                        <div className="flex flex-col md:flex-row md:items-center space-y-4 md:space-y-0 md:space-x-6">
                                            <div className="flex flex-col items-end">
                                                <span className={`text-lg font-medium ${
                                                    stock.price > stock.prevPrice ? 'text-green-500' : 'text-red-500'
                                                }`}>
                                                    ₹{stock.price.toFixed(2)}
                                                </span>
                                                <span className={`text-sm ${
                                                    stock.price > stock.prevPrice ? 'text-green-500' : 'text-red-500'
                                                }`}>
                                                    {stock.price > stock.prevPrice ? '+' : ''}
                                                    {(((stock.price - stock.prevPrice) / stock.prevPrice) * 100).toFixed(2)}%
                                                </span>
                                            </div>
        
                                            {selectedStock?.name === stock.name && (
                                                <div className="flex flex-wrap items-center gap-3">
                                                    <input
                                                        type="number"
                                                        min="1"
                                                        value={quantity}
                                                        onChange={handleQuantityChange}
                                                        className="w-24 px-3 py-2 border-2 border-slate-200 rounded-lg focus:outline-none focus:border-indigo-500"
                                                        onClick={e => e.stopPropagation()}
                                                    />
                                                    <button 
                                                        onClick={e => {
                                                            e.stopPropagation();
                                                            buyStock(stock);
                                                        }}
                                                        className="px-6 py-2 bg-indigo-600 text-white font-medium rounded-lg hover:bg-indigo-700 transition-colors"
                                                    >
                                                        Buy
                                                    </button>
                                                    <button 
                                                        onClick={e => {
                                                            e.stopPropagation();
                                                            sellStock(stock);
                                                        }}
                                                        className="px-6 py-2 border-2 border-red-500 text-red-500 font-medium rounded-lg hover:bg-red-50 transition-colors"
                                                    >
                                                        Sell
                                                    </button>
                                                    <button
                                                        onClick={e => {
                                                            e.stopPropagation();
                                                            toggleWatchlist(stock.name);
                                                        }}
                                                        className="px-6 py-2 border-2 border-slate-200 text-slate-600 font-medium rounded-lg hover:bg-slate-50 transition-colors"
                                                    >
                                                        {watchlist.includes(stock.name) ? 'Unwatch' : 'Watch'}
                                                    </button>
                                                </div>
                                            )}
                                        </div>
                                    </div>
        
                                    {selectedStock?.name === stock.name && portfolio.find(p => p.name === stock.name) && (
                                        <div className="mt-4 p-4 bg-slate-50 rounded-lg">
                                            <div className="text-slate-600">
                                                Your position: {portfolio.find(p => p.name === stock.name).quantity} shares
                                                @ avg. ₹{portfolio.find(p => p.name === stock.name).buyPrice.toFixed(2)}
                                            </div>
                                        </div>
                                    )}
                                </div>
                            ))
                        ) : (
                            <div className="text-center py-12 bg-white rounded-xl border border-slate-200">
                                <p className="text-xl text-slate-500">No stocks found matching "{searchTerm}"</p>
                            </div>
                        )}
                    </div>
                </div>
            );
        }
        
        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<StockMarket />);
        </script>
